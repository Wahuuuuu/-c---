#####  <b style="color: #5DD0C8;">7-8 是不是太胖了</b>
```c
#include <stdio.h>

int main()
{

	int Height = 0;
	double Weight = 0;

	scanf_s("%d", &Height);

	if (100 > Height) {
		return 0;
	}
	else if (Height >= 300) {
		return 0;
	}
	else {
		Weight = (Height - 100.0) * 0.9 * 2;
	}

	printf("%4.1f", Weight);

	return 0;
}
```
